Variables :

count: num
Début
    count = 0
    demander ("quel est votre mot")

    fonction replace_star(chaîne de caractére mot) : chaîne de caractére
        count = nombre de caractere de mot
        mot = ""
        while count > 0 
            mot += "*"
            count -= 1
        fin while
        retourner mot
