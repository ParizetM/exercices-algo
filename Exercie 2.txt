Variables 
mot: chaîne de caractere 
sortie: Bool
total_num : num

Debut
    sortie = Faux
    total_num = 0
    i = 0
    tableau_num = []
    while sortie != Vrai
        demander("Donnez un mot :")
        recuperer(mot)
        tableau_num[i] = compter_cara(mot)
        i += 1
        Si minuscule(mot) == "Stop" alors
            sortie = Vrai
        sinon
            sortie = Faux
            afficher(tableau_num)
        fin si
    fin while

    fonction compter_cara (chaîne de caractere mot) : num
        Variable : nb: num 
         nb = nombre de caractere de mot
        retourner nb
    fin fonction
Fin
